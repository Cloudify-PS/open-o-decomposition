tosca_definitions_version: tosca_simple_profile_for_nfv_1_0_0 
tosca_default_namespace:               # Optional. default namespace (schema, types version) 
description: Open-O Use case described using TOSCA simple profile for NFV.
metadata:
	ID:                                  # ID of this Network Service Descriptor
	vendor:                              # Provider or vendor of the Network Service
	version:                             # Version of the Network Service Descriptor 
imports: 
	- tosca_base_type_definition.yaml
topology_template: 
	inputs:
		flavor ID: 
	ThinCPE:
		type: tosca.nodes.nfv.VNF.Router
		requirements:
			virtualLink1: EnterpriseLAN
			virtualLink2: EnterpriseToPOP
		capabilities:
		  lanPort
		  vcpePort

	VCPE:
		type: tosca.nodes.nfv.VNF.Router
		requirements:
			virtualLink1: POPLAN
			virtualLink2: EnterpriseToPOP
		capabilities:
		  thincpePort
		  lanPort

	VFirewall:
		type: tosca.nodes.nfv.VNF.Router
		requirements:
			virtualLink1: POPLAN
			virtualLink2: WAN
		capabilities:
		  securedPort
		  unsecuredPort

	DCGateway:
		type: tosca.nodes.nfv.VNF.Router
		requirements:
			virtualLink1: DCLAN
			virtualLink2: WAN
		capabilities:
		  internalPort
		  externalPort

	DatabaseServer:
		type: tosca.nodes.nfv.VNF.DatabaseServer
		requirements:
			virtualLink1: DCLAN
		capabilities:
			lanPort

	TrafficIn
		type: tosca.nodes.nfv.CP
		requirements:
			virtualLink: EnterpriseLAN
	 
	TrafficOut
		type: tosca.nodes.nfv.CP
		requirements:
			virtualLink: DCLAN
			
	EnterpriseLAN
		type: tosca.nodes.nfv.VL.Elan
		capabilities:
			-virtual_linkable
				occurrences: 2

	EnterpriseToPOP
		type: tosca.nodes.nfv.VL.Eline
		capabilities:
			-virtual_linkable
				occurrences: 2

	POPLAN
		type: tosca.nodes.nfv.VL.Elan
		capabilities:
			-virtual_linkable
				occurrences: 2

	WAN
		type: tosca.nodes.nfv.VL.Elan
		capabilities:
			-virtual_linkable
				occurrences: 2

	DCLAN
		type: tosca.nodes.nfv.VL.Elan
		capabilities:
			-virtual_linkable
				occurrences: 2

	ForwardingPath:
		type: tosca.nodes.nfv.FP 
		description: Main forwarding graph
		properties:
			policy:
		requirements:
			-forwarder: TrafficIn
			-forwarder: ThinCPE
				capability: lanPort
			-forwarder: ThinCPE
				capability: vcpePort
			-forwarder: VCPE
				capability: thincpePort
			-forwarder: VCPE
				capability: lanPort
			-forwarder: VFirewall
				capability: securedPort
			-forwarder: VFirewall
				capability: unsecuredPort
			-forwarder: DCGateway
				capability: externalPort
			-forwarder: DCGateway
				capability: internalPort
			-forwarder: DatabaseServer
				capability: lanPort
			-forwarder: TrafficOut     

	Groups:
		VNFFG:
			type: tosca.groups.nfv.vnffg
			description: Forwarding graph group
			properties:
				vl: [EnterpriseLAN, EnterpriseToPOP, POPLAN, WAN, DCLAN]
				vnf: [ThinCPE, VCPE, VFirewall, DCGateway, DatabaseServer]
			targets: [ForwardingPath]
